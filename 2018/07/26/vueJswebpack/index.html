const VueLoaderPlugin = require('vue-loader/lib/plugin');
module.exports={       //жЁЎеқ—
    entry: './main.js',  //е…ҘеҸЈж–Үд»¶

    output:{              //е°ҶVUEж–Үд»¶зј–иҜ‘иҫ“еҮә
        path: __dirname, //йЎ№зӣ®ж №и·Ҝеҫ„ жҳҜnode.jsдёӯзҡ„дёҖдёӘе…ЁеұҖеҸҳйҮҸ,е®ғжҢҮеҗ‘еҪ“еүҚжү§иЎҢи„ҡжң¬жүҖеңЁзҡ„зӣ®еҪ•

            loader:'babel-loader',       //ES6иҪ¬жҚўпјҢжүҖжңүзҡ„ES6ж–Үд»¶еҠ иҪҪеҸҠиҪ¬жҚў
            exclude: /node_modules/     //жҺ’йҷӨиҝҷдёӘзӣ®еҪ•
        },
        {
            test: /\.css$/,                       //vue-loader@15.*д№ӢеҗҺйҷӨдәҶеҝ…йЎ»еёҰжңүVueLoaderPlugin д№ӢеӨ–пјҢиҝҳйңҖеҸҰеӨ–еҚ•зӢ¬й…ҚзҪ®css-loaderгҖӮ
            use: ['style-loader','css-loader']

    ]
    }
}

зј–еҶҷ.babelrc

babelжҳҜд»Җд№Ҳ?

babelе®ҳзҪ‘жӯЈдёӯй—ҙдёҖиЎҢй»„иүІеӨ§еӯ—еҶҷзқҖвҖңbabel is a javascript compilerвҖқпјҢзҝ»иҜ‘дёҖдёӢе°ұжҳҜbabelжҳҜдёҖдёӘjavascriptиҪ¬иҜ‘еҷЁгҖӮдёәд»Җд№ҲдјҡжңүbabelеӯҳеңЁе‘ўпјҹеҺҹеӣ жҳҜjavascriptеңЁдёҚж–ӯзҡ„еҸ‘еұ•пјҢдҪҶжҳҜжөҸи§ҲеҷЁзҡ„еҸ‘еұ•йҖҹеәҰи·ҹдёҚдёҠгҖӮд»Ҙes6дёәдҫӢпјҢes6дёӯдёәjavascriptеўһеҠ дәҶз®ӯеӨҙеҮҪж•°гҖҒеқ—зә§дҪңз”Ёеҹҹзӯүж–°зҡ„иҜӯжі•е’ҢSymbolгҖҒPromiseзӯүж–°зҡ„ж•°жҚ®зұ»еһӢпјҢдҪҶжҳҜиҝҷдәӣиҜӯжі•е’Ңж•°жҚ®зұ»еһӢе№¶дёҚиғҪеӨҹй©¬дёҠиў«зҺ°еңЁзҡ„жөҸи§ҲеҷЁе…ЁйғЁж”ҜжҢҒпјҢдёәдәҶиғҪеңЁзҺ°жңүзҡ„жөҸи§ҲеҷЁдёҠдҪҝз”Ёjsж–°зҡ„иҜӯжі•е’Ңж–°зҡ„ж•°жҚ®зұ»еһӢпјҢе°ұйңҖиҰҒдҪҝз”ЁдёҖдёӘиҪ¬иҜ‘еҷЁпјҢе°Ҷjavascriptдёӯж–°еўһзҡ„зү№жҖ§иҪ¬дёәзҺ°д»ЈжөҸи§ҲеҷЁиғҪзҗҶи§Јзҡ„еҪўејҸгҖӮbabelе°ұжҳҜеҒҡиҝҷдёӘж–№йқўзҡ„иҪ¬еҢ–е·ҘдҪңгҖӮ
