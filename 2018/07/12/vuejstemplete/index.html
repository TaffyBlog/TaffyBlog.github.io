实现根据条件禁用DOM

示例如下：


var app = new Vue({ 
    el: '#app',
    data: {
        value: 'a'
    }
});


<div id="app">
    <input type="text" :disabled="value=='b'" v-model="value">
</div>

获取设置复选框和单选按钮值

复选框示例

var app = new Vue({ 
    el: '#app',
    data: {
        values:[]
    }
});


<div id="app">
    <input type="checkbox" v-model="values" value="a" />a
    <input type="checkbox" v-model="values" value="b" />b
    <input type="checkbox" v-model="values" value="c" />c
    <input type="checkbox" v-model="values" value="d" />d
    <span>{{values}}</span>
</div>

单选按钮示例

var app = new Vue({ 
    el: '#app',
    data: {
        values:'0'
    }
});


<div id="app">
    <input type="radio" name="sex" v-model="values" value="0" />男
    <input type="radio" name="sex" v-model="values" value="1" />女
    <span>{{values}}</span>
</div>

获取下拉列表值

var app = new Vue({ 
    el: '#app',
    data: {
        items:[
            {text:"老师",value:1},
            {text:"学生",value:2},
            {text:"家长",value:3},
        ],
        selectvalue:1
    }
});

<div id="app">
    <select v-model="selectvalue">
        <option v-for="item in items" :value="item.value">{{item.text}}</option>
    </select> 
    <span>{{selectvalue}}</span>
</div>

常用修饰符

不允许输入空格

<input v-model.trim="message" />

加上trim，输入空格会被自动移除

引用多个样式

数组方式

var app = new Vue({ 
    el: '#app',
    data: {
        aa:'aa-class',
        bb:'bb-class'
    }
});


<div id="app">
    <span :class="[aa,bb]">
</div>

对象方式

var app = new Vue({ 
    el: '#app',
    data: {
        aa:'aa-class',
        bb:'bb-class',
        flag:true
    }
});


<div id="app">
    <span :class="{aa:flag,bb:flag}">
</div>
